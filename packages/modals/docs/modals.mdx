### Success ConfirmModal

<Playground>
{
    () => {
      const [isOpen, setIsOpen] = useState(false);
      const close = () => setIsOpen(false);
      const open = () => setIsOpen(true);
      return (
        <>
          <ButtonPrimary onClick={open} text={"Show Confirmation modal"} />
          <ConfirmModal
            isOpen={isOpen} 
            close={close}
            header={"Accept the change"}
            bodyText={"Your action will be approved"}
            okCb={close}
            okText={"Accept"}
            state={"success"}/>
        </>
      );
    }
}
</Playground>

### Modal with Close icon

Make sure to provide both `close` function and `iconClose={true}`

<Playground>
{
    () => {
      const [isOpen, setIsOpen] = useState(false);
      const confirm = () => setIsOpen(false);
      const close = () => setIsOpen(false);
      const open = () => setIsOpen(true);
      return (
        <>
          <ButtonPrimary onClick={open} text={"Show Modal with custom footer"} />
          <Modal footer={false} headerText={"Information"} isOpen={isOpen} width={"30%"} close={close} iconClose={true}>
            <BodyUI>
              Lorem ipsum dolor sit amet, sonet everti mei in, at his amet simul

              paulo fabulas, mel iriure albucius atomorum ex. At simul aliquando
              sit, mel putent praesent salutatus id.
            </BodyUI>
          </Modal>
        </>
      );
    }
}
</Playground>

{
    () => {
      const [isOpen, setIsOpen] = useState(false);
      const confirm = () => setIsOpen(false);
      const cancel = () => setIsOpen(false);
      const open = () => setIsOpen(true);
      return (
        <>
          <ButtonPrimary onClick={open} text={"Show Modal with custom footer"} />
          <Modal footer={false} isOpen={isOpen} width={"30%"}>
            <BodyUI>
              Lorem ipsum dolor sit amet, sonet everti mei in, at his amet simul
              fabellas. Amet platonem et eos. Pri viris oportere in. Ad per sale
              paulo fabulas, mel iriure albucius atomorum ex. At simul aliquando
              sit, mel putent praesent salutatus id.
            </BodyUI>
            <div style={{marginTop: "32px", textAlign: "center"}}>
              <ButtonSecondary text="Cancel" onClick={cancel} />
              <ButtonPrimary text="OK" onClick={confirm} />
            </div>
          </Modal>
        </>
      );
    }
}
</Playground>
